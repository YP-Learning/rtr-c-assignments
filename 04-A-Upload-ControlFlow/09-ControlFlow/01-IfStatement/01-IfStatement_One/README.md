int main(void)
{
    int ypp_a, ypp_b, ypp_p;

    ypp_a = 9;

    ypp_p = 30;

    printf("\n\n");

    {
        printf("A is Less than B!!! \n\n");
    }

    if (ypp_b != ypp_p)

    }

    printf("Both Comparision Have Been Done!!!\n\n");

}
